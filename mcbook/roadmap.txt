-- Enable UUID generation
create extension if not exists "pgcrypto";

-- clients: one row per business owner. id = auth.users.id
create table if not exists public.clients (
  id              uuid primary key references auth.users(id) on delete cascade,
  business_name   text not null default '',
  email           text not null default '',
  created_at      timestamptz not null default now()
);

-- services: what the business offers (Haircut, Color Treatment, etc.)
create table if not exists public.services (
  id              uuid primary key default gen_random_uuid(),
  client_id       uuid not null references public.clients(id) on delete cascade,
  name            text not null,
  description     text,
  duration_mins   int  not null check (duration_mins > 0),
  price           numeric(10,2) not null check (price >= 0),
  noshow_fee      numeric(10,2) not null default 0 check (noshow_fee >= 0),
  active          boolean not null default true,
  created_at      timestamptz not null default now()
);
create index if not exists services_client_id_idx on public.services(client_id);
create index if not exists services_active_idx    on public.services(client_id, active);

-- availability_rules: one row per day-of-week per client (0=Sun … 6=Sat)
create table if not exists public.availability_rules (
  id              uuid primary key default gen_random_uuid(),
  client_id       uuid not null references public.clients(id) on delete cascade,
  day_of_week     int  not null check (day_of_week between 0 and 6),
  enabled         boolean not null default true,
  start_time      time not null default '09:00',
  end_time        time not null default '17:00',
  unique (client_id, day_of_week)
);
create index if not exists avail_client_idx on public.availability_rules(client_id);

-- booking_settings: slot duration, advance window, min notice — one row per client
create table if not exists public.booking_settings (
  id                    uuid primary key default gen_random_uuid(),
  client_id             uuid not null references public.clients(id) on delete cascade,
  slot_duration_mins    int  not null default 30,
  advance_window_weeks  int  not null default 4,
  min_notice_hours      int  not null default 2,
  created_at            timestamptz not null default now(),
  unique (client_id)
);

-- blocked_dates: holidays, training days — no bookings accepted
create table if not exists public.blocked_dates (
  id          uuid primary key default gen_random_uuid(),
  client_id   uuid not null references public.clients(id) on delete cascade,
  date        date not null,
  label       text,
  created_at  timestamptz not null default now()
);
create index if not exists blocked_dates_client_date_idx
  on public.blocked_dates(client_id, date);

-- customers: end-customers who book via the widget
create table if not exists public.customers (
  id                  uuid primary key default gen_random_uuid(),
  client_id           uuid not null references public.clients(id) on delete cascade,
  name                text not null,
  email               text not null,
  phone               text,
  stripe_customer_id  text,
  created_at          timestamptz not null default now()
);
create index if not exists customers_client_id_idx   on public.customers(client_id);
create index if not exists customers_email_idx       on public.customers(client_id, email);
create index if not exists customers_created_at_idx  on public.customers(client_id, created_at);

-- bookings: core record. status: scheduled | completed | noshow | cancelled
create table if not exists public.bookings (
  id                  uuid primary key default gen_random_uuid(),
  client_id           uuid not null references public.clients(id) on delete cascade,
  customer_id         uuid not null references public.customers(id),
  service_id          uuid not null references public.services(id),
  date                date not null,
  time                time not null,
  status              text not null default 'scheduled'
                        check (status in ('scheduled','completed','noshow','cancelled')),
  payment_method_id   text,
  noshow_charged      boolean not null default false,
  created_at          timestamptz not null default now()
);
create index if not exists bookings_client_id_idx   on public.bookings(client_id);
create index if not exists bookings_date_idx        on public.bookings(client_id, date);
create index if not exists bookings_status_idx      on public.bookings(client_id, status);
create index if not exists bookings_customer_id_idx on public.bookings(customer_id);
create index if not exists bookings_created_at_idx  on public.bookings(client_id, created_at);

-- payments: audit log of Stripe no-show charges
create table if not exists public.payments (
  id                uuid primary key default gen_random_uuid(),
  booking_id        uuid not null references public.bookings(id),
  client_id         uuid not null references public.clients(id),
  stripe_charge_id  text,
  amount            numeric(10,2) not null,
  type              text not null default 'noshow_fee',
  created_at        timestamptz not null default now()
);
create index if not exists payments_client_id_idx  on public.payments(client_id);
create index if not exists payments_booking_id_idx on public.payments(booking_id);

Access rules:

  clients       — all operations: owner only (auth.uid() = id)

  services      — SELECT: public anon (widget reads without auth)
                — INSERT/UPDATE/DELETE: owner only (auth.uid() = client_id)

  availability_rules — SELECT: public anon
                     — INSERT/UPDATE/DELETE: owner only

  booking_settings   — SELECT: public anon
                     — INSERT/UPDATE/DELETE: owner only

  blocked_dates      — SELECT: public anon
                     — INSERT/DELETE: owner only

  customers     — SELECT/UPDATE/DELETE: owner only (auth.uid() = client_id)
                — INSERT: allow anon (widget creates customers without auth)

  bookings      — SELECT/UPDATE/DELETE: owner only
                — INSERT: allow anon (widget submits bookings without auth)

  payments      — SELECT: owner only
                — INSERT: service role only (Edge Functions use service role)
                — UPDATE/DELETE: deny all

Write CREATE POLICY statements for every table and every operation.
Name policies as: "<table> <operation> <who>"  e.g. "services select public"
Do not include DROP POLICY statements.
